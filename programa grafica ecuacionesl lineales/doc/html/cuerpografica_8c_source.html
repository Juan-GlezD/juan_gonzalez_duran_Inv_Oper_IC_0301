628 char const * gnuplot_tmpfile(gnuplot_ctrl * handle)
629 {
630  static char const * tmp_filename_template = "gnuplot_tmpdatafile_XXXXXX";
631  char * tmp_filename = NULL;
632  int tmp_filelen = strlen(tmp_filename_template);
633 
634 #ifndef WIN32
635  int unx_fd;
636 #endif // #ifndef WIN32
637 
638  assert(handle->tmp_filename_tbl[handle->ntmp] == NULL);
639 
640  /* Open one more temporary file? */
641  if (handle->ntmp == GP_MAX_TMP_FILES - 1) {
642  fprintf(stderr,
643  "maximum # of temporary files reached (%d): cannot open more",
644  GP_MAX_TMP_FILES) ;
645  return NULL;
646  }
647 
648  tmp_filename = (char*) malloc(tmp_filelen+1);
649  if (tmp_filename == NULL)
650  {
651  return NULL;
652  }
653  strcpy(tmp_filename, tmp_filename_template);
654 
655 #ifdef WIN32
656  if (_mktemp(tmp_filename) == NULL)
657  {
658  return NULL;
659  }
660 #else // #ifdef WIN32
661  unx_fd = mkstemp(tmp_filename);
662  if (unx_fd == -1)
663  {
664  return NULL;
665  }
666  close(unx_fd);
667 
668 #endif // #ifdef WIN32
669 
670  handle->tmp_filename_tbl[handle->ntmp] = tmp_filename;
671  handle->ntmp ++;
672  return tmp_filename;
673 }
